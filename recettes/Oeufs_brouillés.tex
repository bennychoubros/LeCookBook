\begin{recipe}
[%
	preparationtime = {\unit[5]{min}}, %h, min, ...
	bakingtime={\unit[15]{min}}, %h, min, ...
	portion = {\portion{2}},
	source = {Bennychou}
]
{Oeufs brouillés}

	%    \graph
	%    {% pictures
	%        small=pic/glass,     % small picture
	%        big=pic/ingredients  % big picture
	%     }

	\ingredients{%
		6 Œufs\\
		3 cuillères à soupe de lait\\
		40g de beurre\\
		poivre\\
		sel\\
	}

	\preparation{%
		\begin{enumerate}
			\item Dans une casserole, casser les œufs puis ajouter le lait, la moitié du beurre, le sel et le poivre.
			\item Faites cuire à feu très doux (pendant environ 12~15min) sans cesser de fouetter les œufs jusqu’à ce qu’ils prennent la consistance d’une crème.
			\item Retirer du feu. Ajouter le reste du beurre en continuant de fouetter pendant qu'il fond complètement.
			\item Servez aussitôt. 
		\end{enumerate}
	}

	\hint{%
		Avec une pincée de piment d'Espelette... Et un verre de Gamay.
	}

\end{recipe}
