\begin{recipe}
[%
	preparationtime = {\unit[15]{min}}, %h, min, ...
	bakingtime={\unit[30]{min}}, %h, min, ...
	portion = {\portion{2}}
]
{Soupe au Panais et Celeri}

	\ingredients{
		2 panais\\
		1 carotte\\
		3 branches de celeri\\
		10cl de lait de coco\\
		15g de Noix de Pécan, ou Noix de Cajou\\
		1 bouillon cube\\
		1 cuillère à café de raz-el-hanout\\
		1 pincée de Piment d'Espelette\\
		Gros Sel\\
	}

	\preparation{%
		\begin{enumerate}
			\item Peler et couper en dés panais, carotte et céleri.
			\item Les placer dans une cocotte, et ajouter 1L d'eau. Ajouter le bouillon cube, le raz-el-hanout et le Piment d'Espelette. Saler à votre goût. Portez à ébullition.
			\item Quand l'eau bout, diminuez le feu et laissez cuire 25 à 30 minutes à feu moyen (eau frémissante).
			\item Pendant ce temps, cassez les noix en petits morceaux.
			\item Quand l'eau est presque évaporée, retirez du feu. Ajoutez le lait de coco et mixez la soupe. Servez dans des bols et ajoutez les morceaux de noix. Servez chaud.
		\end{enumerate}
	}

\end{recipe}
